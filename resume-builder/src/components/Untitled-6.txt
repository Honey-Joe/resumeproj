    <script>
        // Add subtle floating animation to sections on mouse move
        document.addEventListener('mousemove', (e) => {
            const x = (window.innerWidth / 2 - e.pageX) / 25;
            const y = (window.innerHeight / 2 - e.pageY) / 25;
            
            const profile = document.querySelector('.profile-section');
            const content = document.querySelector('.content-section');
            
            if (profile) {
                profile.style.transform = `rotateY(${-x}deg) rotateX(${y}deg)`;
            }
            
            if (content) {
                content.style.transform = `rotateY(${x}deg) rotateX(${y}deg)`;
            }
        });
        
        // Reset rotation when mouse leaves
        document.querySelector('.resume-container').addEventListener('mouseleave', () => {
            const profile = document.querySelector('.profile-section');
            const content = document.querySelector('.content-section');
            
            if (profile) {
                profile.style.transform = 'rotateY(-5deg)';
            }
            
            if (content) {
                content.style.transform = 'rotateY(5deg)';
            }
        });
    </script>
